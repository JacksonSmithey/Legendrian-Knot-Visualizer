from sympy import symbols, diff, Eq, solve

# Define the variables
x, y, z = symbols('x y z')

# Define the equations (as an example, you can modify these)
eq1 = Eq(x**2 + y**2 - 25, 0)
eq2 = Eq(x - 2*y + z, 0)

# Solve the system of equations
solution = solve((eq1, eq2), (x, y, z))

# Print the solutions
print("Solutions:")
for sol in solution:
    print(sol)
